ALTER PROCEDURE [dbo].[spPagoExtra]
	-- Add the parameters for the stored procedure here
	@idPaymentExtra int,
	@saldo int	
AS
BEGIN	
	select pe.amount as pagoInicial ,concat(e.first_name , ' ', e.last_name) as fullname ,p.amount as pago, p.no_document , CONCAT(em.first_name, ' ', em.last_name) as NombreUsuario, tp.description , @saldo as saldo
	from extra_payments pe 
	inner join students e on pe.id_student = e.id
	inner join payments p on pe.id_payment = p.id
	inner join payment_types tp on pe.id_type_payment = tp.id
	inner join users u on p.id_user = u.id  
	inner join employees em on u.id_employee = em.id
	where p.id = @idPaymentExtra;
END
